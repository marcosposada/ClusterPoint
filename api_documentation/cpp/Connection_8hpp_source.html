161  std::vector<unsigned char> reply = socket->send(message);
162  if (this->debug)
163  std::cout << "Response:\n" << std::string(reply.begin(), reply.end()) << std::endl;
164  return new ResponseType(std::string(reply.begin(), reply.end()));
165  } else {

168  std::string frame = "";
169

173  frame += pb.toString();
174 
175  frame = header(frame.size()) + frame;
176 
177  // Send request and get reply
178  std::vector<unsigned char> reply = socket->send(frame);
179
